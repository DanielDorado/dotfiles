(defun my/projectile-ripgrep-regex ()
  "Send arg like with C-u"
  (interactive)
  (let ((current-prefix-arg t)
        (case-fold-search nil))
    (call-interactively 'projectile-ripgrep)))


(use-package ripgrep
  :ensure t
  :config
  ;; This is used to remove --fixed-strings that is inserted for the ripgrep package
  (advice-add 'ripgrep-regexp :around
              (lambda (orig &rest args)
                (let ((case-fold-search nil))
                  (apply orig args)))
              )
  (defun my/ripgrep-regexp-from-dir ()
    "Run `ripgrep-regexp' using the directory of the current buffer or Dired."
    (interactive)
    (let* ((dir (if (derived-mode-p 'dired-mode)
                    (dired-current-directory)
                  default-directory))
           (regexp (read-from-minibuffer
                    (format "Ripgrep regexp in %s for: " (abbreviate-file-name dir))
                    (thing-at-point 'symbol))))
      (let ((case-fold-search nil)) ;; case-sensitive by default
        (ripgrep-regexp regexp dir))))

  )

  (defun my/org-pomodoro-toggle ()
    "Start a pomodoro (and clock in) or, if active, kill it and stop the clock.

Behavior:
- No active pomodoro -> ensure we're clocked in, then start a pomodoro.
- Active pomodoro     -> kill the pomodoro and clock out."
    (interactive)
    (require 'org)
    (require 'org-pomodoro)

    (if (and (fboundp 'org-pomodoro-active-p)
             (org-pomodoro-active-p))
        ;; Pomodoro is running: stop everything
        (progn
          (when (org-clocking-p)
            ;; (org-clock-out nil t): no note prompt, resolves to now
            (org-clock-out nil t)
            (ignore-errors (org-pomodoro-kill))
            ))
      ;; No pomodoro: make sure we're clocking, then start one
      (progn
        ;; If not already clocking, try to clock into current heading.
        ;; If point isn't on a heading, Org will prompt you to pick a task.
        (org-pomodoro)))
    )
